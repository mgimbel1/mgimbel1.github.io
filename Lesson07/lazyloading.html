const images=document.querySelectorAll("[data-src]")]; function preloadImage(img){ const src= img.getAttribute("data-src"); if(!src){ return; } img.src=src; } const imgOptions={ threshold:0, rootMargin:"0px 0px 300px 0px" }; const imgObserver = new IntersectionObserver((enteries, imgObserver) => { enteries.forEach(entry=>{ if(!entry.isIntersecting){ return; }else{ } preloadImage(entry.target); imgObserver.unObserve(entry.target); }) },imgOptions) images.ForEach(image => { imgObserver.Observe(image); });
